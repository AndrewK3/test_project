git status :will show changes in folder
git add 'filename'  : -> will add file to staging
git add .  : -> adds all untracked files to staging 
git commit -m 'message'  : commit staging to local repository
git commit -a 'message'  : commit all to local repository

git fetch : <- get to staging
git dif : check changes
git merge : <- get staging to local file

git log : show what happened
git push -u origin master  : -> push commited changes to github
git push  : -> can be used afterwards to do same push style

touch newfile.txt : creates file on local side
git pull origin master  : pull changed files down
